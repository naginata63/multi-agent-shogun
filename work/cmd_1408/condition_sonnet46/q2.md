## 統合設計案

### 設計原則
- **単一責任の維持**: write（書き込み）とwatch（監視）は異なる実行ライフサイクル
  - inbox_write: 1回呼び出し → 終了（短命プロセス）
  - inbox_watcher: 常駐ループ（長命プロセス）
  - → **完全マージは不適切**。共有ライブラリ方式が正解

- **Python化**: YAML操作の信頼性向上のため、inline python3 を整理されたモジュールに集約

### inbox_core.py の詳細設計

```python
# scripts/lib/inbox_core.py

import fcntl, json, os, tempfile, yaml
from pathlib import Path
from contextlib import contextmanager

SPECIAL_TYPES = ("clear_command", "model_switch", "cli_restart")

@contextmanager
def inbox_lock(inbox_path: Path):
    """排他ロック（flock）コンテキストマネージャ"""
    lock_path = inbox_path.with_suffix(".yaml.lock")
    with open(lock_path, "w") as lock_file:
        fcntl.flock(lock_file, fcntl.LOCK_EX | fcntl.LOCK_NB)
        try:
            yield
        finally:
            fcntl.flock(lock_file, fcntl.LOCK_UN)

def load_inbox(inbox_path: Path) -> dict:
    with open(inbox_path, encoding="utf-8") as f:
        data = yaml.safe_load(f) or {}
    if "messages" not in data:
        data["messages"] = []
    return data

def save_inbox(inbox_path: Path, data: dict):
    fd, tmp_path = tempfile.mkstemp(dir=inbox_path.parent, suffix=".tmp")
    try:
        with os.fdopen(fd, "w") as f:
            yaml.dump(data, f, default_flow_style=False, allow_unicode=True, indent=2)
        os.replace(tmp_path, inbox_path)
    except:
        os.unlink(tmp_path)
        raise

def write_message(inbox_path: Path, msg_id: str, from_: str,
                  timestamp: str, type_: str, content: str):
    """メッセージ書き込み（排他ロック付き、最大3リトライ）"""
    for attempt in range(3):
        try:
            with inbox_lock(inbox_path):
                data = load_inbox(inbox_path)
                data["messages"].append({
                    "id": msg_id, "from": from_, "timestamp": timestamp,
                    "type": type_, "content": content, "read": False,
                })
                # overflow protection: keep max 50
                if len(data["messages"]) > 50:
                    unread = [m for m in data["messages"] if not m.get("read")]
                    read = [m for m in data["messages"] if m.get("read")]
                    data["messages"] = unread + read[-30:]
                save_inbox(inbox_path, data)
                return True
        except BlockingIOError:
            import time; time.sleep(1)
    return False

def get_unread_info(inbox_path: Path) -> dict:
    """unread件数＋special検出（atomicに処理）"""
    data = load_inbox(inbox_path)
    messages = data["messages"]
    unread = [m for m in messages if not m.get("read")]
    specials = [m for m in unread if m.get("type") in SPECIAL_TYPES]

    if specials:
        for m in messages:
            if not m.get("read") and m.get("type") in SPECIAL_TYPES:
                m["read"] = True
        save_inbox(inbox_path, data)

    normal = [m for m in unread if m.get("type") not in SPECIAL_TYPES]
    return {
        "count": len(normal),
        "has_task_assigned": any(m.get("type") == "task_assigned" for m in normal),
        "specials": [{"type": m.get("type"), "content": m.get("content")} for m in specials],
    }

def get_unread_count_fast(inbox_path: Path) -> int:
    """ロックなし高速カウント（fast-path）"""
    try:
        data = load_inbox(inbox_path)
        return sum(1 for m in data["messages"] if not m.get("read"))
    except Exception:
        return 0
```

### 薄くなったinbox_write.sh

```bash
#!/usr/bin/env bash
# inbox_write.sh — inbox_core.py のラッパー
set -e
TARGET="$1"; CONTENT="$2"; TYPE="${3:-wake_up}"; FROM="${4:-unknown}"

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
INBOX="$SCRIPT_DIR/queue/inbox/${TARGET}.yaml"

[ ! -f "$INBOX" ] && { mkdir -p "$(dirname "$INBOX")"; echo "messages: []" > "$INBOX"; }

MSG_ID="msg_$(date +%Y%m%d_%H%M%S)_$(head -c 4 /dev/urandom | xxd -p)"
TIMESTAMP=$(date "+%Y-%m-%dT%H:%M:%S")

"$SCRIPT_DIR/.venv/bin/python3" "$SCRIPT_DIR/scripts/lib/inbox_core.py" write \
    --inbox "$INBOX" --msg-id "$MSG_ID" --from "$FROM" \
    --timestamp "$TIMESTAMP" --type "$TYPE" --content "$CONTENT"

touch "$INBOX" 2>/dev/null  # inotifywait wake-up

[ "$TYPE" = "cmd_new" ] && bash "$(dirname "$0")/automation/cmd_rag_hook.sh" &
"$SCRIPT_DIR/.venv/bin/python3" "$SCRIPT_DIR/scripts/lib/inbox_core.py" \
    update-cmd-status --yaml "$SCRIPT_DIR/queue/shogun_to_karo.yaml" \
    --type "$TYPE" --content "$CONTENT"
"$SCRIPT_DIR/.venv/bin/python3" "$SCRIPT_DIR/scripts/lib/inbox_core.py" \
    update-task-done --tasks-dir "$SCRIPT_DIR/queue/tasks" \
    --type "$TYPE" --from "$FROM" --content "$CONTENT"
```
